## 一、设计决策

### 1.1 真实 LLM，不 Mock

Mock LLM 测不出 JSON 结构、字段名、嵌套层级等问题——这些才是 LLM 调用环节最频繁的 bug 来源。

### 1.2 观测模式，不逐个调用 Agent

触发完整 Temporal Workflow → 被动监听 NATS + 轮询 DB + 查询 Temporal → 自动审批 Gate → 流程走完生成报告。不模拟、不拦截、不注入。

### 1.3 Truth Spec 作为验证基线

不用被测系统源码推导验证规则——代码写错了推导结果也会错。用人维护的 **truth-spec.yaml** 定义"系统应该如何运行"，运行时将实际行为与 Spec 逐项比对。业务架构变更时，先更新 Spec，再改代码。

### 1.4 三种使用模式

| 模式 | 入口 | 说明 |
|------|------|------|
| **Web Dashboard** | `http://172.27.78.109:8500` | 点击运行、实时看流转图、展开每步详情 |
| **CLI 全流程** | `python cli.py run --title "..." --desc "..."` | 命令行触发 + 终端实时输出 |
| **CLI 诊断** | `python cli.py diagnose --req-id "..."` | 对已有 req 做数据一致性快照检查，不触发任何 Agent |

## 四、Observer 观测引擎

### 4.1 核心类

```python
# observer.py

class PipelineObserver:
    """三通道观测引擎: NATS + DB 轮询 + Temporal 查询"""

    def __init__(self, req_id: str, workflow_id: str, gate_strategy: str,
                 truth_spec: dict, db_pool, temporal_client):
        self.req_id = req_id
        self.workflow_id = workflow_id
        self.gate_strategy = gate_strategy  # "auto" | "manual"
        self.spec = truth_spec
        self.db_pool = db_pool
        self.temporal_client = temporal_client
        self.verifier = RuntimeVerifier(truth_spec, db_pool)
        self.timeline: list[dict] = []
        self.findings: list[dict] = []
        self._current_state = "DRAFT"
        self._stop = asyncio.Event()
        self._started_at: datetime | None = None
        self._event_queue: asyncio.Queue = asyncio.Queue()  # SSE 推送队列

    async def run(self) -> RunResult:
        self._started_at = datetime.now(timezone.utc)
        self._append_event("workflow_started", state="DRAFT",
                           data={"workflow_id": self.workflow_id})

        async with asyncio.TaskGroup() as tg:
            tg.create_task(self._listen_nats())
            tg.create_task(self._poll_db())
            tg.create_task(self._poll_temporal())
            if self.gate_strategy == "auto":
                tg.create_task(self._auto_approve_gates())
            tg.create_task(self._watchdog())  # 总体超时保护

        return self._build_result()

    # ── NATS 监听 ──
    async def _listen_nats(self):
        nc = await nats.connect("nats://localhost:4222")
        js = nc.jetstream()

        async def on_context_ready(msg):
            data = json.loads(msg.data.decode())
            payload = data.get("payload", {})
            if payload.get("req_id") != self.req_id:
                return
            state = payload.get("state", "")
            agent = payload.get("agent_id", "")

            # 快照 DB
            db_snap = await self._snapshot_db()

            # 记录事件
            self._append_event("context_built", state=state, agent=agent,
                               data={"context_snapshot": payload, "db_snapshot": db_snap})

            # 运行验证
            f1 = await self.verifier.check_upstream_visibility(state, payload, db_snap)
            self._add_findings(f1)

            await self._event_queue.put({"type": "context_built", "state": state, "agent": agent})

        async def on_agent_result(msg):
            data = json.loads(msg.data.decode())
            if data.get("req_id") != self.req_id:
                return
            agent = data.get("agent_id", "")
            result = data.get("result", {})
            state = self._current_state

            self._append_event("agent_completed", state=state, agent=agent,
                               data={"result": result})

            # 运行验证
            f2 = await self.verifier.check_agent_output_fields(agent, result)
            f3 = await self.verifier.check_data_quantity_constraints(agent, result)
            f4 = await self.verifier.check_persistence_contracts(agent, self.db_pool, self.req_id)
            f5 = await self.verifier.check_llm_audit_completeness(agent, self.req_id)
            self._add_findings(f2 + f3 + f4 + f5)

            await self._event_queue.put({"type": "agent_completed", "state": state, "agent": agent})

        await js.subscribe("context.ready.>", cb=on_context_ready,
                           stream="AI_NATIVE_EVENTS")
        await js.subscribe("agent.result.>", cb=on_agent_result,
                           stream="AI_NATIVE_EVENTS")
        await self._stop.wait()
        await nc.close()

    # ── DB 轮询 ──
    async def _poll_db(self):
        while not self._stop.is_set():
            try:
                row = await self.db_pool.fetchrow(
                    "SELECT status, spec, current_gate FROM requirements WHERE id=$1",
                    self.req_id
                )
                if row:
                    new_state = row["status"]
                    if new_state != self._current_state:
                        old = self._current_state
                        self._current_state = new_state
                        self._append_event("state_transition", state=new_state,
                                           data={"from": old, "to": new_state,
                                                 "db_snapshot": self._flatten_spec(row["spec"])})
                        await self._event_queue.put({"type": "state_change", "state": new_state})
            except Exception:
                pass
            await asyncio.sleep(3)

    # ── Temporal 查询 ──
    async def _poll_temporal(self):
        while not self._stop.is_set():
            try:
                handle = self.temporal_client.get_workflow_handle(self.workflow_id)
                progress = await handle.query("get_progress")
                self._append_event("temporal_heartbeat", data=progress)
                if progress.get("state") in ("done", "blocked"):
                    self._stop.set()
            except Exception:
                pass
            await asyncio.sleep(5)

    # ── Gate 自动审批 ──
    async def _auto_approve_gates(self):
        while not self._stop.is_set():
            try:
                row = await self.db_pool.fetchrow(
                    "SELECT id, gate FROM gate_approvals "
                    "WHERE req_id=$1 AND status='pending' ORDER BY gate LIMIT 1",
                    self.req_id
                )
                if row:
                    async with httpx.AsyncClient() as http:
                        await http.post(
                            f"http://localhost:8000/api/approvals/{row['id']}/approve"
                        )
                    self._append_event("gate_auto_approved",
                                       gate=row["gate"],
                                       data={"gate_id": str(row["id"])})
                    await self._event_queue.put(
                        {"type": "gate_approved", "gate": row["gate"]}
                    )
            except Exception:
                pass
            await asyncio.sleep(5)

    # ── 总体超时保护 ──
    async def _watchdog(self):
        # 从 spec 计算最大允许时间: 所有 agent timeout + 所有 gate sla+grace
        total_seconds = 3600 * 2  # 默认 2h
        await asyncio.sleep(total_seconds)
        if not self._stop.is_set():
            self._append_event("workflow_timeout", data={"limit_s": total_seconds})
            self._stop.set()

    # ── 内部方法 ──
    def _append_event(self, event_type: str, **kwargs):
        self.timeline.append({
            "seq": len(self.timeline) + 1,
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "event_type": event_type,
            **kwargs,
        })

    def _add_findings(self, findings: list[dict]):
        self.findings.extend(findings)

    async def _snapshot_db(self) -> dict:
        row = await self.db_pool.fetchrow(
            "SELECT title, status, spec->'artifacts' as artifacts, "
            "spec->'openapi' as openapi, spec->'erd' as erd "
            "FROM requirements WHERE id=$1", self.req_id
        )
        return self._flatten_spec(row) if row else {}

    def _build_result(self) -> dict:
        # 收尾检查
        self._add_findings(self.verifier.check_flow_contracts_sync(self.timeline))
        self._add_findings(self.verifier.check_gate_progression_sync(self.timeline))
        self._add_findings(self.verifier.check_duplicate_dispatch_sync(self.timeline))
        self._add_findings(self.verifier.check_worktree_cleanup_sync())
        self._add_findings(self.verifier.check_gate_sla_sync(self.timeline))

        return {
            "run_id": f"run-{self._started_at.strftime('%Y%m%d-%H%M%S')}",
            "req_id": self.req_id,
            "workflow_id": self.workflow_id,
            "started_at": self._started_at.isoformat(),
            "finished_at": datetime.now(timezone.utc).isoformat(),
            "final_state": self._current_state,
            "total_duration_s": (datetime.now(timezone.utc) - self._started_at).total_seconds(),
            "gate_strategy": self.gate_strategy,
            "timeline": self.timeline,
            "findings": self.findings,
        }
```

## 五、验证引擎

### 5.1 RuntimeVerifier

所有验证方法分两类：async（需要查 DB）和 sync（只分析时间线）。Observer 在收到每个事件时立即调用对应的验证方法。

```python
# checks/runtime_verifier.py

class RuntimeVerifier:
    def __init__(self, truth_spec: dict, db_pool):
        self.spec = truth_spec
        self.db_pool = db_pool

    # ═══════════════════════════════════════════════════════════════
    # 检查 1: 上游产物可见性
    # 验证: Truth Spec 说 state 的上游是 [A1, A4]，
    #       context 中就必须有对应的 openapi_hint / erd_hint / artifact_context
    # ═══════════════════════════════════════════════════════════════
    async def check_upstream_visibility(
        self, state: str, context_package: dict, db_snapshot: dict
    ) -> list[dict]:
        state_lower = state.lower()
        expected_upstream = self.spec["data_flow"]["upstream_artifacts"].get(state_lower, [])
        context_map = self.spec["data_flow"]["context_key_mapping"]
        findings = []

        for agent in expected_upstream:
            mappings = context_map.get(agent, [])
            if not isinstance(mappings, list):
                mappings = [mappings]
            if not mappings:
                continue

            for mapping in mappings:
                context_path = mapping.get("context_path", "")
                actual = _get_by_path(context_package, context_path)
                is_empty = actual is None or (
                    isinstance(actual, dict) and not actual
                ) or (isinstance(actual, list) and not actual)

                if is_empty:
                    source_path = mapping.get("source", "")
                    db_data = _get_by_path(db_snapshot, source_path)
                    db_has = db_data is not None and db_data != {}

                    findings.append({
                        "rule": "UPSTREAM_ARTIFACT_MISSING_IN_CONTEXT",
                        "severity": "error" if db_has else "warning",
                        "state": state,
                        "agent": agent,
                        "message": (
                            f"期望 {agent} → context.{context_path}"
                            + (f"。DB {source_path} 有数据但未传递 → build_context 桥接问题。"
                               if db_has else
                               f"。DB {source_path} 也无数据 → 上游 {agent} 可能未产出。")
                        ),
                        "context_snapshot": _safe_snapshot({context_path: actual}),
                        "db_snapshot": _safe_snapshot({source_path: db_data}),
                    })
        return findings

    # ═══════════════════════════════════════════════════════════════
    # 检查 2: Agent 输出字段完整性
    # 验证: hard 约束必须满足 (error)，soft 约束通常满足 (warning)
    #       stub agent 仅查 status=completed
    # ═══════════════════════════════════════════════════════════════
    async def check_agent_output_fields(self, agent_id: str, output: dict) -> list[dict]:
        findings = []
        stubs = self.spec.get("agents", {}).get("stubs", [])
        if agent_id in stubs:
            if output.get("status") != "completed":
                findings.append({
                    "rule": "STUB_AGENT_NOT_COMPLETED",
                    "severity": "warning",
                    "agent": agent_id,
                    "message": f"{agent_id} stub agent 返回 status={output.get('status')}",
                })
            return findings

        hard = self.spec.get("constraints", {}).get("hard", {})
        field_key = f"{agent_id.lower()}_output_required_fields"
        for path in hard.get(field_key, []):
            value = _get_by_path(output, path)
            if value is None or (isinstance(value, (list, dict, str)) and not value):
                findings.append({
                    "rule": "AGENT_OUTPUT_MISSING_REQUIRED_FIELD",
                    "severity": "error",
                    "agent": agent_id,
                    "message": f"{agent_id} 输出缺少必填字段: {path}",
                })
        return findings

    async def check_data_quantity_constraints(self, agent_id: str, output: dict) -> list[dict]:
        findings = []
        soft = self.spec.get("constraints", {}).get("soft", {})
        if agent_id == "A4":
            actual = output.get("erd_tables", 0)
            if actual < soft.get("a4_min_erd_tables", 1):
                findings.append({
                    "rule": "A4_ERD_TABLE_COUNT_LOW",
                    "severity": "warning",
                    "message": f"A4 仅生成 {actual} 个表，通常 ≥ {soft['a4_min_erd_tables']}",
                })
        if agent_id == "A6":
            dag = output.get("dag", {})
            actual = len(dag.get("nodes", []))
            if actual < soft.get("a6_min_dag_nodes", 5):
                findings.append({
                    "rule": "A6_DAG_NODE_COUNT_LOW",
                    "severity": "warning",
                    "message": f"A6 仅生成 {actual} 个 DAG 节点，通常 ≥ {soft['a6_min_dag_nodes']}",
                })
        return findings

    # ═══════════════════════════════════════════════════════════════
    # 检查 3: 持久化契约
    # 验证: A4 写入 spec.openapi + api_schemas + spec.erd + erd_designs
    #       其他 Agent 写入 spec.artifacts.{agent_id}
    # ═══════════════════════════════════════════════════════════════
    async def check_persistence_contracts(self, agent_id: str, db_pool, req_id: str) -> list[dict]:
        findings = []
        contracts = self.spec.get("data_flow", {}).get("persistence_contracts", {})
        agent_contracts = contracts.get(agent_id, [])
        default_contract = contracts.get("default", [])

        for c in agent_contracts:
            target = c["target"]
            if target.startswith("spec."):
                field = target.split(".", 1)[1]
                row = await db_pool.fetchrow(
                    "SELECT spec->>$1 as val FROM requirements WHERE id=$2", field, req_id
                )
                if not row or not row["val"]:
                    findings.append({
                        "rule": "PERSISTENCE_CONTRACT_VIOLATED",
                        "severity": "error", "agent": agent_id,
                        "message": f"{agent_id} 应写入 {target}，但值为空",
                    })
            elif target == "api_schemas":
                row = await db_pool.fetchrow(
                    "SELECT COUNT(*) as c FROM api_schemas WHERE req_id=$1", req_id
                )
                if not row or row["c"] == 0:
                    findings.append({
                        "rule": "PERSISTENCE_CONTRACT_VIOLATED",
                        "severity": "error", "agent": agent_id,
                        "message": f"{agent_id} 应写入 api_schemas 表，但无记录",
                    })
            elif target == "erd_designs":
                row = await db_pool.fetchrow(
                    "SELECT COUNT(*) as c FROM erd_designs WHERE req_id=$1", req_id
                )
                if not row or row["c"] == 0:
                    findings.append({
                        "rule": "PERSISTENCE_CONTRACT_VIOLATED",
                        "severity": "error", "agent": agent_id,
                        "message": f"{agent_id} 应写入 erd_designs 表，但无记录",
                    })

        if not agent_contracts and default_contract:
            for dc in default_contract:
                field = dc["target"].replace("{agent_id}", agent_id).split(".", 1)[1]
                row = await db_pool.fetchrow(
                    "SELECT spec->>$1 as val FROM requirements WHERE id=$2", field, req_id
                )
                if not row or not row["val"]:
                    findings.append({
                        "rule": "PERSISTENCE_CONTRACT_VIOLATED",
                        "severity": "warning", "agent": agent_id,
                        "message": f"{agent_id} 应写入 spec.artifacts.{agent_id}，但为空",
                    })
        return findings

    # ═══════════════════════════════════════════════════════════════
    # 检查 4: Gate 推进
    # ═══════════════════════════════════════════════════════════════
    def check_gate_progression_sync(self, timeline: list[dict]) -> list[dict]:
        findings = []
        for gate in self.spec.get("gates", []):
            level = gate["level"]
            expected_next = gate["next_state"]
            approvals = [e for e in timeline
                         if e.get("event_type") in ("gate_approved", "gate_auto_approved")
                         and e.get("gate") == level]
            for approval in approvals:
                idx = timeline.index(approval)
                changes = [e for e in timeline[idx+1:idx+6]
                           if e.get("event_type") in ("state_changed", "state_transition")]
                if changes:
                    actual = changes[0].get("new_state", changes[0].get("state"))
                    if actual != expected_next:
                        findings.append({
                            "rule": "GATE_PROGRESSION_MISMATCH",
                            "severity": "error",
                            "message": f"Gate {level}: 期望 → {expected_next}，实际 → {actual}",
                        })
                else:
                    findings.append({
                        "rule": "GATE_APPROVAL_NO_STATE_CHANGE",
                        "severity": "error",
                        "message": f"Gate {level} 审批后 5 个事件内无状态变更",
                    })
        return findings

    # ═══════════════════════════════════════════════════════════════
    # 检查 5: 流程契约 (Rework / Inner Loop)
    # ═══════════════════════════════════════════════════════════════
    def check_flow_contracts_sync(self, timeline: list[dict]) -> list[dict]:
        findings = []
        contracts = self.spec.get("flow_contracts", {})

        # Rework
        rework = contracts.get("rework", {})
        keyword = rework.get("expect_in_context", "")
        des_ctxs = [e for e in timeline
                    if e.get("state") == "DESIGNING" and e.get("event_type") == "context_built"]
        if len(des_ctxs) >= 2 and keyword:
            last = json.dumps(des_ctxs[-1].get("data", {}).get("context_snapshot", {}))
            if keyword not in last:
                findings.append({
                    "rule": "REWORK_FEEDBACK_MISSING",
                    "severity": "error",
                    "message": f"发生 rework 但第二次 DESIGNING context 无 {keyword}",
                })

        # Inner Loop
        inner = contracts.get("inner_loop", {})
        keyword = inner.get("expect_in_context", "")
        dev_ctxs = [e for e in timeline
                    if e.get("state") == "DEVELOPING" and e.get("event_type") == "context_built"]
        if len(dev_ctxs) >= 2 and keyword:
            last = json.dumps(dev_ctxs[-1].get("data", {}).get("context_snapshot", {}))
            if keyword not in last:
                findings.append({
                    "rule": "INNER_LOOP_FEEDBACK_MISSING",
                    "severity": "error",
                    "message": f"发生 inner loop 但第二次 DEVELOPING context 无 {keyword}",
                })
        return findings

    # ═══════════════════════════════════════════════════════════════
    # 补充检查
    # ═══════════════════════════════════════════════════════════════

    def check_duplicate_dispatch_sync(self, timeline: list[dict]) -> list[dict]:
        """同一 state 在 N 秒内被 dispatch 多次"""
        findings = []
        window = self.spec.get("additional_checks", {}).get("duplicate_dispatch_window_seconds", 300)
        seen = {}
        for e in timeline:
            if e.get("event_type") == "context_built":
                state = e.get("state")
                ts = e.get("timestamp", "")
                if state in seen:
                    prev = seen[state]
                    if ts and prev:
                        try:
                            d = (datetime.fromisoformat(ts) - datetime.fromisoformat(prev)).total_seconds()
                            if d < window:
                                findings.append({
                                    "rule": "DUPLICATE_DISPATCH",
                                    "severity": "warning",
                                    "message": f"状态 {state} 在 {d:.0f}s 内 dispatch 了两次",
                                })
                        except Exception:
                            pass
                seen[state] = ts
        return findings

    def check_gate_sla_sync(self, timeline: list[dict]) -> list[dict]:
        findings = []
        sla_config = self.spec.get("gate_sla", {})
        now = datetime.now(timezone.utc)
        for e in timeline:
            if e.get("event_type") == "gate_created":
                gate = e.get("gate", 0)
                key = f"gate_{gate}"
                sla_h = sla_config.get(key, {}).get("sla_hours", 4)
                approved = any(a for a in timeline
                               if a.get("event_type") in ("gate_approved", "gate_auto_approved")
                               and a.get("gate") == gate)
                if not approved:
                    try:
                        ts = datetime.fromisoformat(e.get("timestamp", ""))
                        if (now - ts).total_seconds() / 3600 > sla_h:
                            findings.append({
                                "rule": "GATE_SLA_EXCEEDED",
                                "severity": "warning",
                                "message": f"Gate {gate} 等待超过 SLA ({sla_h}h)",
                            })
                    except Exception:
                        pass
        return findings

    async def check_llm_audit_completeness(self, agent_id: str, req_id: str) -> list[dict]:
        enabled = self.spec.get("additional_checks", {}).get("llm_audit_completeness", False)
        if not enabled:
            return []
        log_dir = Path("/opt/ai-native/logs/llm_calls") / req_id[:8]
        if log_dir.exists() and not any(agent_id in f.name for f in log_dir.glob("*.json")):
            return [{
                "rule": "NO_LLM_AUDIT_RECORD",
                "severity": "warning", "agent": agent_id,
                "message": f"{agent_id} 已执行但在 llm_calls 中无审计记录",
            }]
        return []

    def check_worktree_cleanup_sync(self) -> list[dict]:
        findings = []
        cfg = self.spec.get("additional_checks", {})
        path = Path(cfg.get("worktree_path", "/tmp/a9-runtimes"))
        max_age = cfg.get("worktree_max_age_minutes", 120)
        if path.exists():
            now = time.time()
            for d in path.iterdir():
                if d.is_dir() and (now - d.stat().st_mtime) / 60 > max_age:
                    findings.append({
                        "rule": "WORKTREE_LEAK",
                        "severity": "warning",
                        "message": f"Worktree {d.name} 超过 {max_age} 分钟未清理",
                    })
        return findings

def _get_by_path(obj: dict, path: str):
    if not path: return None
    current = obj
    for part in path.split("."):
        if isinstance(current, dict):
            current = current.get(part)
        else:
            return None
    return current

def _safe_snapshot(obj, depth=3, max_str=500):
    if depth <= 0: return "..."
    if isinstance(obj, dict):
        return {k: _safe_snapshot(v, depth-1, max_str) for k, v in list(obj.items())[:20]}
    if isinstance(obj, list):
        return [_safe_snapshot(v, depth-1, max_str) for v in obj[:10]]
    if isinstance(obj, str) and len(obj) > max_str:
        return obj[:max_str] + "..."
    return obj
```

# checks/truth_spec_self_check.py

def validate_truth_spec(spec: dict) -> list[dict]:
    issues = []
    agents = spec["agents"]
    state_agents = agents["state_agent_map"]
    parallel = agents.get("parallel_states", {})
    transitions = spec["state_machine"]["normal_flow"]
    gates = spec["gates"]
    upstream = spec["data_flow"]["upstream_artifacts"]
    context_map = spec["data_flow"]["context_key_mapping"]
    stubs = set(agents.get("stubs", []))
    all_states = set(state_agents.keys()) | set(parallel.keys())

    # Gate runs_in_state 合法性 & next_state 可达性
    for g in gates:
        runs_in, next_s = g.get("runs_in_state"), g.get("next_state")
        if runs_in not in all_states:
            issues.append({"rule": "GATE_RUNS_IN_UNKNOWN", "message": f"Gate {g['level']}: {runs_in} 不在 agents 中"})
        if next_s not in transitions:
            issues.append({"rule": "GATE_NEXT_STATE_UNKNOWN", "message": f"Gate {g['level']}: {next_s} 不在 transitions 中"})
        elif next_s not in transitions.get(runs_in, []):
            issues.append({"rule": "GATE_NEXT_STATE_UNREACHABLE", "message": f"Gate {g['level']}: {runs_in} 无法转到 {next_s}"})

    # upstream_artifacts 的 Agent 在 context_key_mapping 中有定义
    all_upstream = set()
    for lst in upstream.values(): all_upstream.update(lst)
    for a in all_upstream:
        if a not in context_map:
            issues.append({"rule": "UPSTREAM_NO_MAPPING", "message": f"upstream {a} 无 context_key_mapping"})

    # transition 的下游状态存在
    for s, nxt in transitions.items():
        for ns in nxt:
            if ns not in transitions:
                issues.append({"rule": "UNDEFINED_STATE", "message": f"{s} → {ns}，{ns} 未定义"})

    # 并行/普通不重叠
    for s in parallel:
        if s in state_agents:
            issues.append({"rule": "STATE_DUPLICATE", "message": f"{s} 同时在 state_agent_map 和 parallel_states"})

    # stub 与 state_agent_map 重叠 → warning
    for stub in stubs:
        for s, a in state_agents.items():
            if a == stub:
                issues.append({"rule": "STUB_IN_STATE_MAP", "severity": "warning", "message": f"{stub} 是 stub 但执行 {s}"})

    # Gate SLA + grace 不超 24h
    for g in gates:
        total = g.get("sla_hours", 0) + (g.get("grace_hours") or 0)
        if total > 24:
            issues.append({"rule": "GATE_WAIT_LONG", "severity": "warning", "message": f"Gate {g['level']} 总等待 {total}h"})

    # persistence_contracts 路径合法
    for agent, contracts in spec.get("data_flow", {}).get("persistence_contracts", {}).items():
        if agent == "default": continue
        for c in contracts:
            t = c.get("target", "")
            if t.startswith("spec.") and t.count(".") > 2:
                issues.append({"rule": "INVALID_PERSISTENCE_TARGET", "message": f"{agent}: {t}"})

    return issues
```

## 六、测试数据清理

### 6.1 设计原则

1. **标记而非猜测** — 所有测试数据以 `TEST-` 前缀的 `external_id` 标识，精确删除，不依赖模糊条件
2. **多级清理保障** — 正常完成自动清 + 异常 crash 残留清 + 手动清 + 定时清，四个层级
3. **先查后删，记录日志** — 每次清理前先 `SELECT COUNT(*)`，清理后记录删除了多少行，可审计
4. **不删业务数据** — 只删 `external_id LIKE 'TEST-%'` 的行，误删概率为零
5. **可配置保留** — `--keep-data` 标志跳过自动清理，方便人工检查测试产出

  external_id = "TEST-{timestamp}-{random8}"
  例: "TEST-20260707-143022-a3f2b1c9"

关联表的外键通过 req_id 关联，所以只需要 external_id 标记 requirements 表，
其余表通过 JOIN 或子查询定位。
```

### 6.3 涉及清理的数据范围

#### 数据库（6 张表）

| 表 | 关联方式 | 内容 |
|----|----------|------|
| `requirements` | `external_id LIKE 'TEST-%'` | 测试创建的 requirement 主记录 |
| `gate_approvals` | `req_id IN (SELECT id FROM requirements WHERE external_id LIKE 'TEST-%')` | Gate 审批记录 |
| `agent_activities` | 同上 | Agent 活动审计日志 |
| `api_schemas` | 同上 | A4 生成的 API schema 版本 |
| `erd_designs` | 同上 | A4 生成的 ERD 设计版本 |
| `test_executions` | 同上 | A11 测试执行结果 |

#### Temporal（工作流记录）

| 资源 | 操作 | 说明 |
|------|------|------|
| Workflow Execution | `terminate` + 等待关闭 | 如果测试中途停止，可能有残留的 running workflow |

Temporal 的工作流历史在 retention 到期后自动清理（默认 3 天），测试工具只需 terminate 正在运行的测试 workflow，不需要删历史记录。

#### 文件系统（3 类）

| 路径 | 内容 | 清理条件 |
|------|------|----------|
| `/tmp/a9-runtimes/wt-*` | A9 的 git worktree 目录 | 创建超过 120 分钟 |
| `/opt/ai-native/logs/llm_calls/{req_id[:8]}/*.json` | 本次测试的 LLM 调用审计日志 | 测试完成且 `--keep-data` 未设置 |
| `/opt/ai-native/logs/llm_audit.jsonl` | LLM 审计元数据行 | 不清理（追加写入，无测试/业务区分） |

注意：`llm_audit.jsonl` 是全局追加写入，不区分测试/业务，**不清理**。`llm_calls/` 按 `req_id[:8]` 分目录，可以精确删。

#### NATS（消息）

**需要清理，但不能直接删消息。** JetStream 不支持按条件（如 req_id）删除消息，只能按 sequence number 或 purge 整个 stream。直接 purge 整个 stream 会影响正在运行的非测试 workflow。

**策略：靠时间窗口 + 去重机制保证安全，而不是删除消息。**

```
风险分析:
  1. Agent 消费到测试消息 → ack → 消息从 stream 删除（interest-based retention）
     ✓ 正常路径，不会有残留
  
  2. Agent 消费到测试消息 → crash 没 ack → 消息回到 stream
     → Worker 重启 → 重新消费此消息 → req_id 在 DB 中还是有效的
     → 但如果此时 DB 已经清掉 → Agent 报错
     ⚠ 需要通过延迟清理来解决
  
  3. Agent 消费到测试消息 → ack → 消息删了
     → Agent 执行过程中，Orchestrator 又 dispatch 了下游 Agent 的消息
     → 下游消息还在 stream 中 → 但 workflow 已 DONE
     → 如果 DB 已清 → 下游 Agent 消费时报错
     ⚠ 需要通过延迟清理来解决
```

**解决方案：延迟清理 + 孤儿兜底，三层防护：**

```
Layer 1 — 延迟清理（主防护）
  流程结束 → 等待 5 分钟（覆盖 base_worker 的 5 分钟去重窗口）
  → 此时所有 NATS 消息已被消费并 ack
  → 清理 DB
  
Layer 2 — Agent 侧防御（意外路径）
  Agent 的 execute() 中如果查 DB 发现 req_id 不存在
  → 静默跳过，仅记录 debug 日志，不报错
  
Layer 3 — 孤儿清理兜底（最终保障）
  超过 24h 的残留手动清理 → startup_cleanup()
```

# cleanup.py

import asyncio
import asyncpg
import logging
import shutil
from pathlib import Path
from datetime import datetime, timezone, timedelta

logger = logging.getLogger(__name__)

TEST_EXTERNAL_ID_PREFIX = "TEST-%"

# ── 数据库清理 ──

async def cleanup_test_requirement(db_pool, req_id: str) -> dict:
    """清理单个测试 requirement 的所有关联数据。返回清理统计。"""
    stats = {}

    async with db_pool.acquire() as conn:
        # 1. 先验证这是测试数据
        row = await conn.fetchrow(
            "SELECT external_id FROM requirements WHERE id=$1", req_id
        )
        if not row:
            return {"error": f"req_id {req_id} 不存在"}
        if not (row["external_id"] or "").startswith(TEST_EXTERNAL_ID_PREFIX):
            return {"error": f"req_id {req_id} 不是测试数据 (external_id={row['external_id']})，拒绝删除"}

        # 2. 逐表删除（按外键依赖顺序）
        tables = [
            ("test_executions",  "req_id"),
            ("gate_approvals",   "req_id"),
            ("agent_activities", "req_id"),
            ("api_schemas",      "req_id"),
            ("erd_designs",      "req_id"),
        ]

        for table, col in tables:
            result = await conn.execute(
                f"DELETE FROM {table} WHERE {col}=$1", req_id
            )
            # asyncpg 的 DELETE 返回格式是 "DELETE N"
            count = int(result.split()[-1]) if result else 0
            if count > 0:
                stats[table] = count
                logger.info(f"清理 {table}: {count} 行")

        # 3. 最后删主表
        result = await conn.execute(
            "DELETE FROM requirements WHERE id=$1", req_id
        )
        count = int(result.split()[-1]) if result else 0
        stats["requirements"] = count

    return {"req_id": req_id, "cleaned": True, "tables": stats}


async def cleanup_all_test_data(db_pool) -> dict:
    """清理所有 TEST- 前缀的 requirement 及其关联数据。"""
    stats = {}

    async with db_pool.acquire() as conn:
        # 先统计
        total = await conn.fetchval(
            "SELECT COUNT(*) FROM requirements WHERE external_id LIKE $1",
            f"{TEST_EXTERNAL_ID_PREFIX}%"
        )
        if total == 0:
            return {"cleaned": False, "message": "没有找到测试数据", "total_found": 0}

        # 获取所有测试 req_id
        rows = await conn.fetch(
            "SELECT id, external_id FROM requirements WHERE external_id LIKE $1",
            f"{TEST_EXTERNAL_ID_PREFIX}%"
        )
        test_ids = [r["id"] for r in rows]

        # 逐表删除
        tables = [
            ("test_executions",  "req_id"),
            ("gate_approvals",   "req_id"),
            ("agent_activities", "req_id"),
            ("api_schemas",      "req_id"),
            ("erd_designs",      "req_id"),
        ]

        for table, col in tables:
            for tid in test_ids:
                result = await conn.execute(f"DELETE FROM {table} WHERE {col}=$1", tid)
                count = int(result.split()[-1]) if result else 0
                stats[table] = stats.get(table, 0) + count

        # 删主表
        result = await conn.execute(
            "DELETE FROM requirements WHERE external_id LIKE $1",
            f"{TEST_EXTERNAL_ID_PREFIX}%"
        )
        stats["requirements"] = int(result.split()[-1]) if result else 0

    logger.info(f"一键清理完成: 共清理 {total} 个测试 requirement, 详情: {stats}")
    return {"cleaned": True, "total_requirements": total, "tables": stats}


async def cleanup_orphan_test_data(db_pool, max_age_hours: int = 24) -> dict:
    """清理超过 max_age_hours 的测试残留数据。

    应对场景: 测试工具崩溃或被 kill -9，Observer 的 finally 块没执行。
    服务器启动时自动调用此函数。
    """
    cutoff = datetime.now(timezone.utc) - timedelta(hours=max_age_hours)

    async with db_pool.acquire() as conn:
        rows = await conn.fetch(
            "SELECT id, external_id, created_at FROM requirements "
            "WHERE external_id LIKE $1 AND created_at < $2",
            f"{TEST_EXTERNAL_ID_PREFIX}%", cutoff,
        )

        if not rows:
            return {"cleaned": False, "message": f"没有超过 {max_age_hours}h 的孤儿测试数据"}

        cleaned = 0
        for row in rows:
            result = await cleanup_test_requirement(db_pool, row["id"])
            if result.get("cleaned"):
                cleaned += 1

        return {
            "cleaned": True,
            "orphans_found": len(rows),
            "orphans_cleaned": cleaned,
            "max_age_hours": max_age_hours,
        }


# ── 文件系统清理 ──

def cleanup_worktrees(worktree_base: str = "/tmp/a9-runtimes",
                      max_age_minutes: int = 120) -> dict:
    """清理过期的 A9 worktree 目录。"""
    path = Path(worktree_base)
    if not path.exists():
        return {"cleaned": False, "message": f"目录不存在: {worktree_base}"}

    now = datetime.now().timestamp()
    cleaned = []
    errors = []
    total_size_freed = 0

    for wt_dir in path.iterdir():
        if not wt_dir.is_dir():
            continue
        if not wt_dir.name.startswith("wt-a9rt-") and not wt_dir.name.startswith("wt-"):
            continue  # 不是 A9 worktree，跳过

        age_min = (now - wt_dir.stat().st_mtime) / 60
        if age_min > max_age_minutes:
            try:
                dir_size = sum(f.stat().st_size for f in wt_dir.rglob("*") if f.is_file())
                shutil.rmtree(wt_dir)
                cleaned.append({"name": wt_dir.name, "age_minutes": round(age_min)})
                total_size_freed += dir_size
                logger.info(f"清理 worktree: {wt_dir.name} (age={age_min:.0f}min, size={dir_size/1024:.0f}KB)")
            except Exception as e:
                errors.append({"name": wt_dir.name, "error": str(e)})
                logger.warning(f"清理 worktree 失败: {wt_dir.name}: {e}")

    return {
        "cleaned": len(cleaned) > 0,
        "removed_count": len(cleaned),
        "errors": errors,
        "total_size_freed_kb": round(total_size_freed / 1024),
        "details": cleaned,
    }


def cleanup_llm_call_logs(req_id: str, log_base: str = "/opt/ai-native/logs/llm_calls") -> dict:
    """删除指定 req_id 的 LLM 调用审计日志目录。"""
    req_dir = Path(log_base) / req_id[:8]
    if not req_dir.exists():
        return {"cleaned": False, "message": f"目录不存在: {req_dir}"}

    # 只删该 req_id 前缀的文件
    deleted = 0
    for f in req_dir.glob(f"*{req_id[:8]}*.json"):
        f.unlink()
        deleted += 1

    # 如果目录空了，删除目录
    if not any(req_dir.iterdir()):
        req_dir.rmdir()

    return {"cleaned": deleted > 0, "files_deleted": deleted, "path": str(req_dir)}


# ── Temporal 清理 ──

async def terminate_test_workflow(temporal_client, workflow_id: str, reason: str = "test cleanup") -> dict:
    """Terminate 一个正在运行的测试 Temporal Workflow。"""
    try:
        handle = temporal_client.get_workflow_handle(workflow_id)
        desc = await handle.describe()
        if desc.status in (1, 2):  # RUNNING or CONTINUED_AS_NEW
            await handle.terminate(reason=reason)
            logger.info(f"Terminated workflow: {workflow_id}")
            return {"terminated": True, "workflow_id": workflow_id}
        else:
            return {"terminated": False, "workflow_id": workflow_id,
                    "message": f"Workflow 状态为 {desc.status}，无需 terminate"}
    except Exception as e:
        logger.warning(f"Terminate workflow {workflow_id} 失败: {e}")
        return {"terminated": False, "workflow_id": workflow_id, "error": str(e)}


# ── 综合清理 ──

async def full_test_cleanup(db_pool, temporal_client, req_id: str,
                            workflow_id: str | None = None,
                            keep_data: bool = False) -> dict:
    """测试完成后的综合清理入口。

    由 Observer 在 finally 块中调用。
    如果 keep_data=True，所有清理跳过。
    """
    if keep_data:
        logger.info(f"keep_data=True，保留测试数据: req_id={req_id}")
        return {"skipped": True, "reason": "keep_data", "req_id": req_id}

    report = {"req_id": req_id, "workflow_id": workflow_id}

    # 1. Terminate Temporal workflow
    if workflow_id:
        report["temporal"] = await terminate_test_workflow(temporal_client, workflow_id)

    # 2. 清理数据库
    report["database"] = await cleanup_test_requirement(db_pool, req_id)

    # 3. 清理 LLM 审计日志
    report["llm_logs"] = cleanup_llm_call_logs(req_id)

    # 4. 清理 A9 worktree
    report["worktrees"] = cleanup_worktrees()

    logger.info(f"测试清理完成: req_id={req_id}, "
                f"db={report['database'].get('cleaned')}, "
                f"llm_logs={report['llm_logs'].get('files_deleted', 0)} files")
    return report
```

# observer.py — PipelineObserver.run() 方法增强

class PipelineObserver:
    async def run(self) -> RunResult:
        self._started_at = datetime.now(timezone.utc)
        try:
            # ... 原有观测逻辑 ...
            self._append_event("workflow_started", ...)
            async with asyncio.TaskGroup() as tg:
                tg.create_task(self._listen_nats())
                tg.create_task(self._poll_db())
                tg.create_task(self._poll_temporal())
                if self.gate_strategy == "auto":
                    tg.create_task(self._auto_approve_gates())
                tg.create_task(self._watchdog())
        except Exception as exc:
            self._append_event("observer_error", data={"error": str(exc)})
            logger.error(f"Observer 异常: {exc}", exc_info=True)
        finally:
            # ── 延迟清理: 等待 NATS 消息消费完毕 ──
            try:
                if not self._keep_data:
                    # 等待 5 分钟，确保所有 NATS 消息已被 Agent 消费并 ack
                    # (匹配 base_worker.py 的 5 分钟去重窗口)
                    logger.info("等待 NATS 消息消费 (300s) 后再清理 DB...")
                    await asyncio.sleep(300)

                cleanup_report = await full_test_cleanup(
                    db_pool=self.db_pool,
                    temporal_client=self.temporal_client,
                    req_id=self.req_id,
                    workflow_id=self.workflow_id,
                    keep_data=self._keep_data,
                )
                self._append_event("cleanup_completed", data=cleanup_report)
            except Exception as cleanup_err:
                logger.error(f"清理失败: {cleanup_err}", exc_info=True)
                self._append_event("cleanup_failed", data={"error": str(cleanup_err)})

        return self._build_result()
```

### 6.9 异常场景处理

| 场景 | 触发条件 | 清理结果 |
|------|----------|----------|
| 正常完成 | 流程到达 DONE 或 BLOCKED | 等待 300s NATS 窗口后 finally → 全部清 |
| 用户手动停止 | 点击 [停止] → `_stop.set()` | 等待 300s NATS 窗口后 finally → 全部清 |
| 进程被 kill -9 | 外部 `kill -9` | DB 孤儿下次启动清。NATS 消息残留 → Worker 消费 → Agent 查 DB 发现 req_id 不存在 → 静默跳过 |
| 下游 Agent 收到旧消息 | DB 已清但 NATS 消息未 ack | Agent 查 DB: req_id 不存在 → debug 日志 + ack + 跳过 |
| 清理中途失败 | DB 连接断开等 | 逐表删除，每步 catch。未删的下次孤儿清理补删 |
| 非测试 req_id | 误传业务 req_id | `external_id` 检查 → 拒绝 |
| 保留数据模式 | `keep_data=True` | 跳过延迟等待，不删任何数据 |
| Temporal 历史 | Workflow execution history | 不清理。namespace retention (默认 3 天) 自动处理 |
| llm_audit.jsonl | 全局 LLM 审计日志 | 不清理（追加写入，不区分测试/业务） |

### 6.10 CLI 清理命令
